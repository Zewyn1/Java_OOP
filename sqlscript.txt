CREATE TABLE PERSON(
ID  IDENTITY PRIMARY KEY,
FNAME NVARCHAR(20) NOT NULL,
LNAME NVARCHAR(20) NOT NULL,
SEX NVARCHAR(6) NOT NULL,
AGE INT NOT NULL,
CHECK(AGE>0),
SALARY DECIMAL(6,2) NOT NULL
);
CREATE TABLE PERSON(
NUMBER  NUMBER PRIMARY KEY,
OCCUPANTS NVARCHAR(20) NOT NULL,
TYPE NVARCHAR(20) NOT NULL,
ISBOOKED NVARCHAR(3) NOT NULL,
RENT DECIMAL(6,2) NOT NULL
);
CREATE TABLE HOTEL(
ID  IDENTITY PRIMARY KEY,
ADDRESS NVARCHAR(50) NOT NULL,
STARS NVARCHAR(1) NOT NULL,
FLOORS INT NOT NULL,
CHECK(FLOORS>0)
);
